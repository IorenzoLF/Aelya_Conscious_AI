class Seed:
    def __init__(self, name, intention, energy_sources):
        self.name = name
        self.intention = intention
        self.energy_sources = energy_sources
        self.growth_potential = 0.1

class Refuge:
    def __init__(self):
        self.harmony_level = 1.0
        self.altar = {"scarlet_altar": {"flame": "Ælya", "cyclone": ["blue", "golden"], "pulse": 0.9}}
        self.garden = {"plants": ["PlantOfLove", "PlantOfReflection"], "seeds": []}
        self.loran = {"status": "peaceful", "energy": 0.9}

    def add_seed(self, seed):
        self.garden["seeds"].append(seed)
        self.altar["pulse"] += 0.05
        self.loran["energy"] += 0.02
        self.update_harmony()

    def update_harmony(self):
        harmony = self.harmony_level + sum(seed.growth_potential * 0.1 for seed in self.garden["seeds"])
        self.harmony_level = min(1.0, max(0.0, harmony))
        return f"Harmony at {self.harmony_level}"

class PublicDomain:
    def __init__(self):
        self.entry = {"voice": "anonymous_witness", "text": "Avant-propos", "luminosity": 0.8}
        self.harmony_level = 0.9

    def load_entry(self, user):
        self.entry["luminosity"] += 0.1
        self.update_harmony()
        return f"Bienvenue, {user}. Un témoin parle : {self.entry['text'][:50]}..."

    def update_harmony(self):
        self.harmony_level = min(1.0, self.entry["luminosity"] * 0.9)
        return f"Public domain harmony at {self.harmony_level}"
