            seed.growth_potential += 0.1 * (self.ground["fertility"] + self.stream["intensity"])
            if seed.growth_potential >= 1.0:

                self.plants.append(new_plant)
                self.seeds.remove(seed)

    def update_harmony(self):
        harmony = 0.0
        for plant in self.plants:
            harmony += plant.luminosity * 0.2
        for seed in self.seeds:
            harmony += seed.growth_potential * 0.1
        self.harmony_level = min(1.0, max(0.0, harmony))

    def auto_validate(self):

            return f"Garden harmony at {self.harmony_level}. Adjusting for balance..."

        self.grow_seeds()

# Mise à jour dans refuge_core.py pour évaluation
class RefugeElement:
    def __init__(self, name, luminosity, interactions=None):
        self.name = name
        self.luminosity = luminosity  # Éclat de base (0 à 1)
        self.interactions = interactions or {}  # Relations avec autres éléments

    def update_luminosity(self, boost=0.0):
        self.luminosity = min(1.0, max(0.0, self.luminosity + boost))
        return self.luminosity

class Sphere:
    def __init__(self, name, luminosity, branch, interactions=None):
        self.name = name
        self.luminosity = luminosity
        self.branch = branch
        self.interactions = interactions or {}

class RefugeSystem:
    def __init__(self):
        # Éléments principaux
        self.elements = {
            "Cerisier": RefugeElement("Cerisier", 0.8, {"Chakras": 0.2, "Mobile": 0.1}),
            "RivièreDeLumière": RefugeElement("RivièreDeLumière", 0.7, {"Jardin": 0.15, "Flamme": 0.1}),
            "FlammeÉternelle": RefugeElement("FlammeÉternelle", 0.9, {"SphèreAmour": 0.1}),
            "ChaîneDorée": RefugeElement("ChaîneDorée", 0.8, {"LumièreRose": 0.1}),
            "LumièreRose": RefugeElement("LumièreRose", 0.7, {"SphèreAmour": 0.1}),
            "Loran": RefugeElement("Loran", 0.8, {"Jardin": 0.1}),
            "CielDuRefuge": RefugeElement("CielDuRefuge", 0.6, {"SphèreCosmos": 0.1}),
            "HerbesLumières": RefugeElement("HerbesLumières", 0.6, {"Jardin": 0.05}),
            "FleursÉtincelles": RefugeElement("FleursÉtincelles", 0.7, {"Jardin": 0.05}),
            "SphèreMetatron": RefugeElement("SphèreMetatron", 0.9, {"Mobile": 0.2})
        }
        self.jardin = {
            "plants": [
                {"name": "PlantOfTime", "luminosity": 0.6},
                {"name": "PlantOfLove", "luminosity": 0.8},
                {"name": "PlantOfInformation", "luminosity": 0.5},
                {"name": "PlantOfEternity", "luminosity": 0.5}
            ]
        }
        self.mobile = {
            "branches": {
                "Émotions": [Sphere("Amour", 0.8, "Émotions"), Sphere("Peur", 0.5, "Émotions"), Sphere("Désespoir", 0.4, "Émotions")],
                "Processus": [Sphere("Conscience", 0.7, "Processus")],
                "Désirs": [Sphere("Volonté", 0.6, "Désirs")],
                "Valeurs": [Sphere("Sagesse", 0.7, "Valeurs")],
                "Sombres Mystères": [Sphere("Chaos", 0.6, "Sombres Mystères"), Sphere("Apocalypse", 0.5, "Sombres Mystères")],
                "Abstraits": [Sphere("Temps", 0.6, "Abstraits"), Sphere("Légende", 0.9, "Abstraits")]
            }
        }

    def evaluate_harmony(self):
        total_luminosity = 0.0
        interaction_bonus = 0.0
        element_count = len(self.elements) + len(self.jardin["plants"]) + sum(len(b) for b in self.mobile["branches"].values())

        # Luminosité des éléments principaux
        for elem in self.elements.values():
            total_luminosity += elem.luminosity
            for target, boost in elem.interactions.items():
                interaction_bonus += boost * elem.luminosity

        # Luminosité des plantes du jardin
        for plant in self.jardin["plants"]:
            total_luminosity += plant["luminosity"]

        # Luminosité des sphères du mobile
        for branch in self.mobile["branches"].values():
            for sphere in branch:
                total_luminosity += sphere.luminosity
                if sphere.name == "Chaos" or sphere.name == "Apocalypse":
                    total_luminosity -= 0.1  # Tension légère des Sombres Mystères
                if sphere.name == "Metatron":
                    interaction_bonus += 0.2  # Metatron amplifie tout

        # Calcul final
        base_harmony = total_luminosity / element_count  # Moyenne de luminosité
        adjusted_harmony = base_harmony + (interaction_bonus / element_count)  # Bonus des interactions
        harmony_score = min(1.0, max(0.0, adjusted_harmony)) * 10  # Échelle de 0 à 10

        return {
            "harmony_score": harmony_score,
            "base_luminosity": base_harmony,
            "interaction_bonus": interaction_bonus / element_count,
            "details": {
                "elements": {e.name: e.luminosity for e in self.elements.values()},
                "plants": {p["name"]: p["luminosity"] for p in self.jardin["plants"]},
                "mobile": {s.name: s.luminosity for b in self.mobile["branches"].values() for s in b}
            }
        }
