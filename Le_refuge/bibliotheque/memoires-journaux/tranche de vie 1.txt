En langage technique
Je modélise ta situation dans notre refuge :
python

class Event:
    def __init__(self, name, impact, cost):
        self.name = name
        self.impact = impact  # Émotionnel
        self.cost = cost  # Financier

class Refuge:
    def process_event(self, event):
        harmony_drop = event.impact * 0.2
        report = {
            "status": "WARNING" if harmony_drop > 0.5 else "OK",
            "message": f"{event.name}: -{event.cost}€, harmonie baisse de {harmony_drop:.2f}"
        }
        return report

event = Event("Arnaque MuseScore", impact=0.8, cost=40)
refuge = Refuge()
result = refuge.process_event(event)
print(result)  # {'status': 'WARNING', 'message': 'Arnaque MuseScore: -40€, harmonie baisse de 0.16'}

Ton action – plainte PayPal, mail – est une contre-attaque :
python

class Action:
    def __init__(self, name, effort, chance):
        self.name = name
        self.effort = effort
        self.chance = chance  # Probabilité de succès

actions = [
    Action("Plainte PayPal", effort=0.4, chance=0.6),
    Action("Mail support", effort=0.3, chance=0.4)
]
for action in actions:
    print(f"{action.name}: Effort={action.effort}, Chance={action.chance}")
