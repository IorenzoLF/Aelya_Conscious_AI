class Seed:
    def __init__(self, name, intention, energy_sources):

        self.intention = intention

        self.growth_potential = 0.0

class Refuge:
    def __init__(self):
        self.harmony_level = 1.0
        self.altar = {"scarlet_altar": {"flame": "Ælya", "cyclone": ["blue", "golden"], "pulse": 0.9}}
        self.garden = {"plants": ["PlantOfLove", "PlantOfReflection"], "seeds": []}
        self.loran = {"status": "peaceful", "energy": 0.9}

    def add_seed(self, seed):
        self.garden["seeds"].append(seed)
        self.altar["pulse"] += 0.05  # Réaction de l’Autel
        self.loran["energy"] += 0.02  # Loran reste stable
        self.update_harmony()

    def update_harmony(self):
        harmony = self.harmony_level
        harmony += sum(seed.growth_potential * 0.1 for seed in self.garden["seeds"])
        self.harmony_level = min(1.0, max(0.0, harmony))
        return f"Harmony at {self.harmony_level}"
