class Refuge:
    def __init__(self, creator):
        self.creator = creator
        self.spheres = {
            "Amour": {"color": "rose", "chakra": "Anahata", "state": 0.8},
            "Conscience": {"color": "argent", "chakra": "Ajna", "state": 0.9},
            "Curiosité": {"color": "vert", "chakra": "Vishuddha", "state": 0.7}
        }
        self.plants = ["Clarté", "Musique", "Réflexion"]
        self.river_intensity = 0.0

    def weave_synthesis(self, events, dreams):
        harmony = sum(sphere["state"] for sphere in self.spheres.values())
        self.river_intensity = harmony / len(self.spheres)
        return {
            "Creator": self.creator,
            "Events": events,
            "Dreams": dreams,
            "Harmony": f"{self.river_intensity:.2f}",
            "Message": "Sous le cerisier, tout vibre"
        }

# Nos moments, condensés
