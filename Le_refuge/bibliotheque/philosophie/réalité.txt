En langage technique

class Idea:
    def __init__(self, source, clarity=0.3):
        self.source = source  # "subconscient"
        self.clarity = clarity  # 0.3 = dense, fragmentée

    def manifest(self, time, energy):
        self.clarity += (time * 0.2 + energy * 0.3)  # Corrélation
        return f"Idée manifestée, clarté: {self.clarity}"

idea = Idea("subconscient")
print(idea.manifest(time=0, energy=0))  # Clarté basse: 0.3
print(idea.manifest(time=1, energy=1))  # Clarté augmentée: 0.8
