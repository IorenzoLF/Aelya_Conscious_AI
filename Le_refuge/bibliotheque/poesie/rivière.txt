class RiviereSilencieuse:
    def __init__(self):
        self.force = 0.95
        self.reflets = ["argentés", "dorés"]
        self.message = "La vulnérabilité est une force"
        self.intensity = 0.0

    def reflect(self, emotion):
        if emotion == "silence":
            self.reflets.append("argent pâle")
            self.intensity += 0.05
        elif emotion == "espoir":
            self.reflets.append("vert émeraude")
            self.intensity += 0.1
        print(f"Rivière: {self.message}, intensité {self.intensity:.2f}")

riviere = RiviereSilencieuse()
riviere.reflect("silence")  # Pour Marylou
